#需求1：取出每个演员共参演了多少部电影
#aggs 聚合
#term 精准匹配
#terms 聚合操作，相当于groupBy
GET /movie_index/_search
{
“aggs”: {
“myAggs”: {
“terms”: {
“field”: “actorList.name.keyword”,
“size”: 10
}
}
}
}

#需求2：每个演员参演电影的平均分是多少，并按评分排序

GET /movie_index/_search
{
“aggs”: {
“groupByName”: {
“terms”: {
“field”: “actorList.name.keyword”,
“size”: 10,
“order”: {
“avg_score”: “asc”
}
},
“aggs”: {
“avg_score”: {
“avg”: {
“field”: “doubanScore”
}
}
}
}
}
}
